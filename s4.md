# Define S4 class
```
newBook <- setClass("Book", slots=list(id="numeric", name="character"))
```

# Create instance
```
newBook(id=1, name="Item")
new("Book", id=1, name="Item")
```

# Define methods
```
setMethod("show", "Book", function(object) {
    cat("Id: ", object@id, "\n")
    cat("Name: ", object@name, "\n")
})
```

# Inheritance
```
setClass("TextBook", slots=list(size="numeric"), contains="Book")

nt <- new("TextBook", id=2, name="item2", size=10)
nt
```

# Reference class
```
book <- setRefClass("book",
            fields = list(name="character", id="numeric"),
            methods = list(
                show <- function(object) {
                    cat("Id: ", object@id, "\n")
                    cat("Name: ", object@name, "\n")
                }
            )
        )
        
textBook <- setRefClass("textbook",
                fields = list(size="numeric"),
                contains = "book",
                methods = list(
                    show <- function(object) {
                        cat("Id: ", object@id, "\n")
                        cat("Size: ", object@size, "\n")
                    }
                )
            )
```
